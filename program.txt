# HLT, ADD, SUB, STO, LDA,
# BR, BRZ, BRP, IN, OUT, DAT


####################		TAKING INPUTS AND RESETTING VALUES

	IN
	STO	n
	STO	temp
	IN
	STO	a
	IN
	STO	b

	LDA	zero
	STO	hundreds
	STO	tens
	STO	units
	STO	outputbool


####################		SEPERATING DIGITS


	BR	sthun

subhun	STO 	temp
	LDA	hundreds
	ADD	one
	STO	hundreds
sthun	LDA	temp
	SUB	hun
	BRP	subhun
	BR	stten

subten	STO 	temp
	LDA	tens
	ADD	one
	STO	tens
stten	LDA	temp
	SUB	ten
	BRP	subten
	LDA	temp
	STO	units
	LDA	outputbool
	BRZ	conv

################### OUTPUT


output	

###################  CONVERTING TO BASE 10

conv	LDA	a
	SUB	ten
	BRZ	next

addhun	LDA	hundreds
	BRZ	addten
	SUB	one
	STO	hundreds
	LDA	tens
	ADD	a
	STO	tens
	BR	addhun

addten	LDA	tens
	BRZ	next
	SUB	one
	STO	tens
	LDA	units
	ADD	a
	STO	units
	BR	addten

next	LDA	units
	STO	n
	STO	temp
	LDA	b
	SUB	ten
	BRZ	output
