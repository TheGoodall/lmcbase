# HLT, ADD, SUB, STO, LDA,
# BR, BRZ, BRP, IN, OUT, DAT


####################		TAKING INPUTS AND RESETTING VALUES

	IN
	STO	n
	STO	temp
	IN
	STO	a
	IN
	STO	b

	LDA	zero
	STO	hundreds
	STO	tens
	STO	units
	STO	outputbool
	BR	sthun


####################		SEPERATING DIGITS


subhun	STO 	temp
	LDA	hundreds
	ADD	one
	STO	hundreds
	SUB	nine
	BRP	error
sthun	LDA	temp
	SUB	hun
	BRP	subhun
	BR	stten

subten	STO 	temp
	LDA	tens
	ADD	one
	STO	tens
stten	LDA	temp
	SUB	ten
	BRP	subten
	LDA	temp
	STO	units


	LDA	outputbool

	BRZ	conv

################### OUTPUT


output	LDA	hundreds
	OUT
	LDA	tens
	OUT
	LDA	units
	OUT
zero	DAT	000
##################		ERROR

error	LDA	nnn
	OUT
one	DAT	001


###################  CONVERTING TO BASE 10




conv	LDA	a
	SUB	ten
	BRZ	next

addhun	LDA	hundreds
	BRZ	addten
	SUB	one
	STO	hundreds
	LDA	tens
	ADD	a
	STO	tens
	BR	addhun

addten	LDA	tens
	BRZ	next
	SUB	one
	STO	tens
	LDA	units
	ADD	a
	STO	units
	BR	addten
